320void PendingConnection::handleAcceptedLogin(shared_ptr<LoginPacket> packet)
321{
322 if(packet->m_ugcPlayersVersion != server->m_ugcPlayersVersion)
323 {
324 // Send the pre-login packet again with the new list of players
325 sendPreLoginResponse();
326 return;
327 }
328
329 // Guests use the online xuid, everyone else uses the offline one
330 PlayerUID playerXuid = packet->m_offlineXuid;
331 if(playerXuid == INVALID_XUID) playerXuid = packet->m_onlineXuid;
332
333 shared_ptr<ServerPlayer> playerEntity = server->getPlayers()->getPlayerForLogin(this, name, playerXuid,packet->m_onlineXuid);
334 if (playerEntity != nullptr)
335 {
336#if defined(_WINDOWS64) && defined(MINECRAFT_SERVER_BUILD)
337 ServerRuntime::ServerLogManager::OnAcceptedPlayerLogin(GetPendingConnectionSmallId(connection), name);
338#endif
339 server->getPlayers()->placeNewPlayer(connection, playerEntity, packet);
340 connection = nullptr; // We've moved responsibility for this over to the new PlayerConnection, nullptr so we don't delete our reference to it here in our dtor
341 }
342 done = true;
