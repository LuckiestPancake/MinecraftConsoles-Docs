17static unsigned int nametagColorForIndex(int index)
18{
19 static const unsigned int s_firstColors[] = {
20 0xff000000, // WHITE (represents the "white" player, but using black as the colour)
21 0xff33cc33, // GREEN
22 0xffcc3333, // RED
23 0xff3333cc, // BLUE
24 0xffcc33cc, // PINK
25 0xffcc6633, // ORANGE
26 0xffcccc33, // YELLOW
27 0xff33dccc // TURQUOISE
28 };
29#ifndef __PSVITA__
30 if (index >= 0 && index < 8) // Use original colors for the first 8 players
31 return s_firstColors[index];
32 if (index >= 8 && index < MINECRAFT_NET_MAX_PLAYERS)
33 {
34 float h = (float)((index * 137) % 360) / 60.f;
35 int i = (int)h;
36 float f = h - i;
37 float q = 1.f - f;
38 float t = 1.f - (1.f - f);
39 float r = 0.f, g = 0.f, b = 0.f;
40 switch (i % 6)
41 {
42 case 0: r = 1.f; g = t; b = 0.f; break;
43 case 1: r = q; g = 1.f; b = 0.f; break;
44 case 2: r = 0.f; g = 1.f; b = t; break;
45 case 3: r = 0.f; g = q; b = 1.f; break;
46 case 4: r = t; g = 0.f; b = 1.f; break;
47 default: r = 1.f; g = 0.f; b = q; break;
48 }
49 int ri = (int)(r * 255.f) & 0xff, gi = (int)(g * 255.f) & 0xff, bi = (int)(b * 255.f) & 0xff;
50 return 0xff000000u | (ri << 16) | (gi << 8) | bi;
51 }
52#endif
53 return 0xFF000000; //Fallback if exceeds 256 somehow
54}
