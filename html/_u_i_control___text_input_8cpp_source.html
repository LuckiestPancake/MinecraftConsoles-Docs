95void UIControl_TextInput::setCaretVisible(bool visible)
96{
97 if (!m_parentScene || !m_parentScene->getMovie())
98 return;
99
100 // Check once whether this SWF's FJ_TextInput actually has a m_mcCaret child.
101 // IggyValuePathMakeNameRef always succeeds (creates a ref to undefined),
102 // so we validate by trying to read a property from the resolved path.
103 if (!m_bCaretChecked)
104 {
105 IggyValuePath caretPath;
106 if (IggyValuePathMakeNameRef(&caretPath, getIggyValuePath(), "m_mcCaret"))
107 {
108 rrbool test = false;
109 IggyResult res = IggyValueGetBooleanRS(&caretPath, m_nameVisible, NULL, &test);
110 m_bHasCaret = (res == 0);
111 }
112 else
113 {
114 m_bHasCaret = false;
115 }
116 m_bCaretChecked = true;
117 }
118 if (!m_bHasCaret)
119 return;
120
121 IggyValuePath caretPath;
122 if (IggyValuePathMakeNameRef(&caretPath, getIggyValuePath(), "m_mcCaret"))
123 {
124 IggyValueSetBooleanRS(&caretPath, m_nameVisible, NULL, visible);

157UIControl_TextInput::EDirectEditResult UIControl_TextInput::tickDirectEdit()
158{
159 if (m_iDirectEditCooldown > 0)
160 m_iDirectEditCooldown--;
161
162 if (!m_bDirectEditing)
163 {
164 setCaretVisible(false);
165 return eDirectEdit_Continue;
166 }
167
168 // Enforce caret visibility and position every tick — setLabel() and Flash
169 // focus changes can reset both at any time.
170 setCaretVisible(true);
171 setCaretIndex(m_iCursorPos);
172
173 // For SWFs without m_mcCaret, insert '_' at the cursor position.
174 // All characters remain visible — '_' sits between them like a cursor.
175 if (!m_bHasCaret)
176 {
177 wstring display = m_editBuffer;
178 display.insert(m_iCursorPos, 1, L'_');
179 setLabel(display.c_str());
180 }
181
182 EDirectEditResult result = eDirectEdit_Continue;
183 bool changed = false;
184
185 // Consume typed characters from the KBM buffer
186 wchar_t ch;
187 while (g_KBMInput.ConsumeChar(ch))
188 {
189 if (ch == 0x08) // Backspace
190 {
191 if (m_iCursorPos > 0)
192 {
193 m_editBuffer.erase(m_iCursorPos - 1, 1);
194 m_iCursorPos--;
195 changed = true;
196 }
197 }
198 else if (ch == 0x0D) // Enter — confirm edit
199 {
200 m_bDirectEditing = false;
201 m_iDirectEditCooldown = 4;
202 setLabel(m_editBuffer.c_str(), true);
203 setCaretVisible(false);
204 return eDirectEdit_Confirmed;
205 }
206 else if (m_iCharLimit <= 0 || (int)m_editBuffer.length() < m_iCharLimit)
207 {
208 m_editBuffer.insert(m_iCursorPos, 1, ch);
209 m_iCursorPos++;
210 changed = true;
211 }
212 }
213
214 // Arrow keys, Home, End, Delete for cursor movement
215 if (g_KBMInput.IsKeyPressed(VK_LEFT) && m_iCursorPos > 0)
216 {
217 m_iCursorPos--;
218 setCaretIndex(m_iCursorPos);
219 }
220 if (g_KBMInput.IsKeyPressed(VK_RIGHT) && m_iCursorPos < (int)m_editBuffer.length())
221 {
222 m_iCursorPos++;
223 setCaretIndex(m_iCursorPos);
224 }
225 if (g_KBMInput.IsKeyPressed(VK_HOME))
226 {
227 m_iCursorPos = 0;
228 setCaretIndex(m_iCursorPos);
229 }
230 if (g_KBMInput.IsKeyPressed(VK_END))
231 {
232 m_iCursorPos = (int)m_editBuffer.length();
233 setCaretIndex(m_iCursorPos);
234 }
235 if (g_KBMInput.IsKeyPressed(VK_DELETE) && m_iCursorPos < (int)m_editBuffer.length())
236 {
237 m_editBuffer.erase(m_iCursorPos, 1);
238 changed = true;
239 }
240
241 // Escape — cancel edit and restore original text
242 if (g_KBMInput.IsKeyPressed(VK_ESCAPE))
243 {
244 m_editBuffer = m_textBeforeEdit;
245 m_bDirectEditing = false;
246 m_iDirectEditCooldown = 4;
247 setLabel(m_editBuffer.c_str());
248 setCaretVisible(false);
249 return eDirectEdit_Cancelled;
250 }
251
252 if (changed)
253 {
254 if (m_bHasCaret)
255 {
256 setLabel(m_editBuffer.c_str());
257 setCaretIndex(m_iCursorPos);
258 }
259 // SWFs without caret: the cursor block above already updates the label every tick
260 }
261
262 return eDirectEdit_Continue;
263}
